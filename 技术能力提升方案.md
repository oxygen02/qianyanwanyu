## 三、核心引擎优化

### 3.1 ink-effect.js 性能优化

**问题**：7层渲染，每层都遍历所有glyphs

**优化方案**：
```javascript
// 预计算哈希值（避免每次重算）
const _glyphHashCache = new Map()
function getGlyphHash(index, charCode) {
  const key = `${index}-${charCode}`
  if (!_glyphHashCache.has(key)) {
    const h1 = ((index * 37 + charCode) % 1000) / 1000
    const h2 = (((h1 * 7919) % 1000) / 1000)
    const h3 = (((h1 * 104729) % 1000) / 1000)
    _glyphHashCache.set(key, { h1, h2, h3 })
  }
  return _glyphHashCache.get(key)
}
```

### 3.2 paper.js 纹理优化

**问题**：每次渲染创建新的 SimplexNoise 和 ImageData

**优化方案**：
```javascript
// 复用噪声实例（带种子控制确保一致性）
class PaperTextureGenerator {
  constructor(seed = 42) {
    this.noise = new SimplexNoise(seed)
    this.grainNoise = new SimplexNoise(seed + 1)
    this.grooveNoise = new SimplexNoise(seed + 2)
  }
}
```

### 3.3 增量渲染策略

**场景**：用户调整单个参数（如字号）

**优化**：避免全量重渲染
```javascript
// 仅更新变化的渲染层
async _updateInkLayer(changes) {
  if (changes.includes('fontSize') || changes.includes('variation')) {
    await this._renderInkBlock()
  } else if (changes.includes('inkOpacity')) {
    await this._updateOpacity()
  }
}
```

### 4.3 工具函数抽象

```javascript
// utils/helpers/debounce.js
function debounce(fn, delay) {
  let timer = null
  return function(...args) {
    clearTimeout(timer)
    timer = setTimeout(() => fn.apply(this, args), delay)
  }
}

// utils/helpers/throttle.js
function throttle(fn, interval) {
  let lastTime = 0
  return function(...args) {
    const now = Date.now()
    if (now - lastTime >= interval) {
      lastTime = now
      fn.apply(this, args)
    }
  }
}
```

### 6.2 性能监控代码

```javascript
// engine/performanceMonitor.js
class PerformanceMonitor {
  constructor() {
    this.marks = new Map()
  }

  start(label) {
    this.marks.set(label, Date.now())
  }

  end(label) {
    const start = this.marks.get(label)
    if (!start) return null
    const duration = Date.now() - start
    this.marks.delete(label)

    if (duration > PERFORMANCE_TARGETS.RENDER_TIME.SINGLE_PAGE) {
      console.warn(`[Performance] ${label} 耗时 ${duration}ms，超过目标`)
    }
    return duration
  }
}

const monitor = new PerformanceMonitor()
```
